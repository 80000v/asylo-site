36 /// An implementation of the SecretSealer abstract interface that binds the
37 /// secrets to the enclave identity on a local machine. The secrets sealed by
38 /// this sealer can only be unsealed on the same machine.
39 ///
40 /// The LocalSecretSealer can be configured to seal secrets either to MRENCLAVE
41 /// or MRSIGNER. The LocalSecretSealer class provides two factory methods--one
42 /// that creates a LocalSecretSelaer configured to seal secrets to MRENCLAVE and
43 /// another that creates a LocalSecretSealer configured to seal secrets to
44 /// MRSIGNER. In the MRENCLAVE-sealing mode, the default SealedSecretHeader
45 /// generated by the sealer binds the secrets to the MRENCLAVE portion of the
46 /// enclave's identity. In the MRSIGNER mode, the default SealedSecretHeader
47 /// generated by the sealer binds the secrets to the MRSIGNER portion of the
48 /// enclave's identity. In either mode, the sealed secret is bound to all bits
49 /// of MISCSELECT, and all security-sensitive bits of ATTRIBUTES, as defined in
50 /// secs_attributes.h.
51 ///
52 /// Note that the SealedSecretHeader provided to the Seal() method also controls
53 /// the cipher suite used for sealing the secret. The default setting of this
54 /// field (as populated by SetDefaultHeader() ) is subject to change. As a
55 /// result, users must not make any assumptions about the default cipher suite.
56 /// If they wish to use a specific cipher suite, they must manually verify or
57 /// override the cipher suite set by the SetDefaultHeader() method.
58 ///
59 /// Sample usage for Seal():
60 /// ```
61 /// std::unique_ptr<LocalSecretSealer> sealer = CreateMrsignerSealer();
62 ///
63 /// SealedSecretHeader header;
64 /// // Fill out the portions of the header that must be set by the client.
65 /// header.set_secret_name("my name");
66 /// header.set_secret_version("my version");
67 /// header.set_secret_purpose("my purpose");
68 /// // secret_handling_policy is a client-specific string, which, for example,
69 /// could be a serialized proto.
70 /// string secret_handling_policy = ...
71 /// header.set_secret_handling_policy(secret_handling_policy);
72 /// sealer->SetDefaultHeader(&header);
73 ///
74 /// // Override fields in the default header, if desired.
75 /// ...
76 ///
77 /// // Generate the secret to be sealed.
78 /// std::vector<uint8_t, CleansingAllocator> secret = ...;
79 ///
80 /// // Generate the additional authenticated data to be tied to the secret.
81 /// string additional_authenticated_data;
82 /// addtional_authenticated_data = ...
83 ///
84 /// // Seal the secret and the additional authenticated data.
85 /// SealedSecret sealed_secret;
86 /// Status status = sealer->Seal(header, additional_authenticated_data,
87 /// secret, &sealed_secret);
88 /// ```
89 ///
90 /// Sample usage for Unseal():
91 /// ```
92 /// std::unique_ptr<LocalSecretSealer> sealer = CreateMrsignerSealer();
93 /// SealedSecret sealed_secret;
94 /// ...
95 /// sealed_secret.ParseFromString(...);
96 ///
97 /// std::vector<uint8_t, CleansingAllocator<uint8_t>> secret;
98 /// Status status = sealer->Unseal(sealed_secret, secret);
99 /// if (!status.ok()) {
100 /// return status;
101 /// }
102 /// // secret now holds the unsealed secret. The policy and
103 /// // additional_authenticated_data in the sealed_secret are now
104 /// // authenticated.
105 /// ```
106 ///
107 /// It should be noted that the LocalSecretSealer's configuration only affects
108 /// the default header generated by the sealer. Users can override the generated
109 /// default header. A sealer in either MRENCLAVE or MRSIGNER configuration can
110 /// unseal secrets that are sealed by a sealer in either configuration.
111 class LocalSecretSealer : public SecretSealer {
112  public:
113  /// Creates a LocalSecretSealer that seals secrets to the MRENCLAVE part of
114  /// the enclave code identity.
115  ///
116  /// \return A smart pointer that owns the created sealer.
117  static std::unique_ptr<LocalSecretSealer> CreateMrenclaveSecretSealer();
118 
119  /// Creates a LocalSecretSealer that seals secrets to the MRSIGNER part of the
120  /// enclave identity.
121  ///
122  /// \return A smart pointer that owns the created sealer.
123  static std::unique_ptr<LocalSecretSealer> CreateMrsignerSecretSealer();
124 
125  LocalSecretSealer(const LocalSecretSealer &other) = delete;
126  virtual ~LocalSecretSealer() = default;
127 
128  LocalSecretSealer &operator=(const LocalSecretSealer &other) = delete;
129 
130  // From SecretSealer interface.
131  SealingRootType RootType() const override;
132  std::string RootName() const override;
133  std::vector<EnclaveIdentityExpectation> RootAcl() const override;
134  Status SetDefaultHeader(SealedSecretHeader *header) const override;
135  Status Seal(const SealedSecretHeader &header,
136  ByteContainerView additional_authenticated_data,
137  ByteContainerView secret, SealedSecret *sealed_secret) override;
138  Status Unseal(const SealedSecret &sealed_secret,
139  CleansingVector<uint8_t> *secret) override;
140 
141  private:
142  // Maximum size (in bytes) of each protected message (including authenticated
143  // data). A protected message may not be larger than 32MB.
144  //
145  // A size-limit of 32MiB (2^25 bytes) allows the cryptor to safely encrypt
146  // 2^48 messages (see https://cyber.biu.ac.il/aes-gcm-siv/). On a 4GHz
147  // single-threaded Intel processor, assuming 1 byte/cycle AES-GCM processing
148  // bandwidth, this yields a key-lifetime of over 2^16 years, if the enclave
149  // did nothing but execute seal/unseal operations 24/7. On a 256-threaded
150  // machine, the key lifetime would reduce to ~256 years.
151  static constexpr size_t kMaxAesGcmSivMessageSize = (1 << 25);
152 
153  // Instantiates LocalSecretSealer that sets client_acl in the default sealed
154  // secret header per |default_client_acl|.
155  LocalSecretSealer(const CodeIdentityExpectation &default_client_acl);
156 
157  // Cryptor to perform AEAD operations.
158  std::unique_ptr<AesGcmSivCryptor> cryptor_;
159 
160  // The default client ACL for this SecretSealer.
161  CodeIdentityExpectation default_client_acl_;
162 };
