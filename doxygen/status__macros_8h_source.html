24 /// Evaluates an expression that produces an `Status`-like object with
25 /// a `.ok()` method. If this method returns false, the object is
26 /// returned from the current function.
27 ///
28 /// Example:
29 /// ```
30 /// ::asylo::Status MultiStepFunction() {
31 /// ASYLO_RETURN_IF_ERROR(Function(args...));
32 /// ASYLO_RETURN_IF_ERROR(foo.Method(args...));
33 /// return ::asylo::Status::OkStatus();
34 /// }
35 /// ```
36 #define ASYLO_RETURN_IF_ERROR(expr) do
37  {
38  const auto _asylo_status_to_verify = (expr);
39  if (ABSL_PREDICT_FALSE(!_asylo_status_to_verify.ok())) {
40  return _asylo_status_to_verify;
41  } \
42 }while (false)

44 /// Evaluates an expression `rexpr` that returns a `StatusOr`-like
45 /// object with `.ok()`, `.status()`, and `.ValueOrDie()` methods. If
46 /// the result is OK, moves its value into the variable defined by
47 /// `lhs`, otherwise returns the result of the `.status()` from the
48 /// current function. The error result of `.status` is returned
49 /// unchanged. If there is an error, `lhs` is not evaluated: thus any
50 /// side effects of evaluating `lhs` will only occur if `rexpr.ok()`
51 /// is true.
52 ///
53 /// Interface:
54 /// ```
55 /// ASYLO_ASSIGN_OR_RETURN(lhs, rexpr)
56 /// ```
57 ///
58 /// Example: Assigning to an existing variable:
59 /// ```
60 /// ValueType value;
61 /// ASYLO_ASSIGN_OR_RETURN(value, MaybeGetValue(arg));
62 /// ```
63 ///
64 /// Example: Assigning to an expression with side effects:
65 /// ```
66 /// MyProto data;
67 /// ASYLO_ASSIGN_OR_RETURN(*data.mutable_str(), MaybeGetValue(arg));
68 /// // No field "str" is added on error.
69 /// ```
70 ///
71 /// Example: Assigning to a `std::unique_ptr`.
72 /// ```
73 /// std::unique_ptr<T> ptr;
74 /// ASYLO_ASSIGN_OR_RETURN(ptr, MaybeGetPtr(arg));
75 /// ```
76 #define ASYLO_ASSIGN_OR_RETURN(lhs, rexpr) do
77  {
78  auto _asylo_status_or_value = (rexpr);
79  if (ABSL_PREDICT_FALSE(!_asylo_status_or_value.ok())) {
80  return _asylo_status_or_value.status();
81  }
82  lhs = std::move(_asylo_status_or_value).ValueOrDie(); \
83 }while (false)
